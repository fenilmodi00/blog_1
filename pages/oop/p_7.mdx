#### 3 Create abstract class called Shape which has three sub classes say Triangle, Rectangle, and Circle. Define one method area() in abstract class and override this area() in these three subclasses to calculate area for specific object i.e. area() of Triangle subclass should calculate area of triangle. Same for rectangle, circle.

```java filename="P7_3.java" copy
import java.util.Scanner;
public class P7_3 {
    public static void main(String[] args) {
        Scanner sc=new Scanner (System.in);
        int l,b;
        double r;
        System.out.println("Enter the length and breath of triangle:- ");
        l=sc.nextInt();
        b=sc.nextInt();
        triangle t=new triangle();
        t.get_info(l,b);
        t.area();
        System.out.println("Enter the length and breath of rectangle:-");
        l=sc.nextInt();
        b=sc.nextInt();
        rectangle R=new rectangle();
        R.get_info(l,b);
        R.area();
        System.out.println("Enter the radius of circle:");
        r = sc.nextDouble();
        circle c = new circle();
        c.get_info(r);
        c.area();
    }
}
abstract class shape{
    int length,breath;
    double radius;
    abstract void area();
}
class triangle extends shape{
    void get_info(int l,int b)
    {
        length=l;
        breath=b;
    }
    void area()
    {
        double area =0.5*length*breath;
        System.out.println("Area of triangle:-"+area);
    }
}
class rectangle extends shape{
    void get_info(int l,int b)
    {
        length=l;
        breath=b;
    }
    void area()
    {
        int area =length*breath;
        System.out.println("Area of rectangle:-"+area);
    }
}
class circle extends shape{
    void get_info(double r)
    {
        radius=r;
    }
    @Override
    void area()
    {
        double area =2*3.14*radius*radius;
        System.out.println("Area of circle:-"+area);
    }
}
```

interface shape
{
    abstract void area();
}
class Rectangle implements shape
{
    int length;
    int breath;
    void get_info(int len, int bre)
    {
        length = len;
        breath = bre;
    }

    {
        System.out.println("Area of Rectangle is : "+ (length*breath));

        height = thei;
        base = tbas;

    {
        System.out.println("Area of Triangle is : "+ (0.5*height*base));

    {
        radius = srad;
